116  void Gradient<C, InterFunc>::computeInterpolation( C * buffer, size_t size, const InterFunc2 & functor, float begin, float end ) const {
117  if ( this -> pointsVector.getSize() == 1 ) {
118  //fill the complete array with the same color
119  for ( size_t i = 0; i < size; i++ ) {
120  buffer[i] = this -> pointsVector[0] -> getColor();
121  }
122 
123  } else if ( this -> pointsVector.getSize() > 1 ) {
124  struct OrderFunctor {
125  bool operator()( const GradientPoint<C> * p1, const GradientPoint<C> * p2 ) const {
126  return ( p1 -> getPosition() < p2 -> getPosition() );
127  }
128  };
129 
130 
131  if ( !this -> isOrdered ) {
132  const_cast< Gradient<C, InterFunc> * >( this ) -> pointsVectorOrdered = const_cast< Gradient<C, InterFunc> * >( this ) -> pointsVector;
133  const_cast< Gradient<C, InterFunc> * >( this ) -> pointsVectorOrdered.sort( OrderFunctor() );
134  //Now we have the point correctly ordered.
135  const_cast< Gradient<C, InterFunc> * >( this ) -> isOrdered = true;
136  }
137  float multiplier = float( size ) / ( end - begin );
138  for ( typename Vector<GradientPoint<C> * >::Size i = 1; i < this -> pointsVectorOrdered.getSize(); i++ ) {
139  const GradientPoint<C> & p1 = *this -> pointsVectorOrdered[i - 1];
140  const GradientPoint<C> & p2 = *this -> pointsVectorOrdered[i];
141 
142  //Now we have to compute the beginning and ending index of the interpolations of the two colors.
143 
144  float startClamped = Math::max<float>( begin, p1.getPosition() );
145  float endClamped = Math::min<float>( end, p2.getPosition() );
146 
147  int startIndex = Math::clamp<int>( int( ( startClamped - begin ) * multiplier ), 0, size );
148  int endIndex = Math::clamp<int>( int( ( endClamped - begin ) * multiplier ), 0, size );
149 
150  float increment = ( 1.0f - ( startClamped - p1.getPosition() + ( p2.getPosition() - endClamped ) ) ) / ( endIndex - startIndex );
151  float x = increment * 0.5f + ( startClamped - p1.getPosition() );
152  for ( int i = startIndex; i < endIndex; i++ ) {
153  buffer[i] = functor( p1.getColor(), p2.getColor(), x );

155  }
