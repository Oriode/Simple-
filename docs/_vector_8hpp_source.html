377 template<typename Compare>
378 void Vector<T>::quicksort( RandomAccessIterator start, RandomAccessIterator end, Compare func ) {
379  // Create an auxiliary stack
380  typedef struct {
381  RandomAccessIterator start;
382  RandomAccessIterator end;
383  } quicksortStackItem;
384 
385  quicksortStackItem * stack = new quicksortStackItem[( end - start ) / 2 + 1];
386 
387 
388  // push initial values of l and h to stack
389  stack[0].start = start;
390  stack[0].end = end;
391 
392  //initialize stack iterator
393  quicksortStackItem * top = &( stack[0] );
394 
395  // Keep popping from stack while is not empty
396  while ( top >= stack ) {
397  end = top -> end;
398  start = top -> start;
399  top--;
400 
401  RandomAccessIterator storeIndex = start;
402 
403  for ( auto i = start; i < end; i++ ) {
404  if ( func( *i, *end ) ) {
405  Vector::swap( i, storeIndex );
406  storeIndex++;
407  }
408  }
409  Vector::swap( end, storeIndex );
410 
411  // If there are elements on left side of pivot, then push
412  if ( storeIndex - 1 > start ) {
413  top++;
414  top -> start = start;
415  top -> end = storeIndex - 1;
416  }
417 
418  // If there are elements on right side of pivot, then push
419  if ( storeIndex + 1 < end ) {
420  top++;
421  top -> start = storeIndex + 1;
422  top -> end = end;
423  }
424  }
425 
426  delete[] stack;
427 }
