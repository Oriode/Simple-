160  MATH_FUNC_QUALIFIER bool clamp( Line<T> * l, const Rectangle<T> & r ) {
161  Line<T> & line = *l;
162 
163  // compute outcodes for P0, P1, and whatever point lies outside the clip rectangle
164  unsigned int outcode0 = r.getZone( line.getP0() );
165  unsigned int outcode1 = r.getZone( line.getP1() );
166  bool accept = false;
167 
168  while ( true ) {
169  if ( !( outcode0 | outcode1 ) ) { // Bitwise OR is 0. Trivially accept and get out of loop
170  accept = true;
171  break;
172  } else if ( outcode0 & outcode1 ) { // Bitwise AND is not 0. Trivially reject and get out of loop
173  break;
174  } else {
175  // failed both tests, so calculate the line segment to clip
176  // from an outside point to an intersection with clip edge
177  Vec2<T> p;
178 
179  // use reference for simpler formulas
180  const Vec2<T> & p0 = line.getP0();
181  const Vec2<T> & p1 = line.getP1();
182 
183  // At least one endpoint is outside the clip rectangle; pick it.
184  unsigned int outcodeOut = outcode0 ? outcode0 : outcode1;
185 
186  // Now find the intersection point;
187  // use formulas y = y0 + slope * (x - x0), x = x0 + (1 / slope) * (y - y0)
188  if ( outcodeOut & ( unsigned int ) Rectangle<T>::Position::Top ) { // point is above the clip rectangle
189  p.x = p0.x + ( p1.x - p0.x ) * ( r.getTop() - p0.y ) / ( p1.y - p0.y );
190  p.y = r.getTop();
191  } else if ( outcodeOut & ( unsigned int ) Rectangle<T>::Position::Bottom ) { // point is below the clip rectangle
192  p.x = p0.x + ( p1.x - p0.x ) * ( r.getBottom() - p0.y ) / ( p1.y - p0.y );
193  p.y = r.getBottom();
194  } else if ( outcodeOut & ( unsigned int ) Rectangle<T>::Position::Right ) { // point is to the right of clip rectangle
195  p.y = p0.y + ( p1.y - p0.y ) * ( r.getRight() - p0.x ) / ( p1.x - p0.x );
196  p.x = r.getRight();
197  } else if ( outcodeOut & ( unsigned int ) Rectangle<T>::Position::Left ) { // point is to the left of clip rectangle
198  p.y = p0.y + ( p1.y - p0.y ) * ( r.getLeft() - p0.x ) / ( p1.x - p0.x );
199  p.x = r.getLeft();
200  }
201 
202  // Now we move outside point to intersection point to clip
203  // and get ready for next pass.
204  if ( outcodeOut == outcode0 ) {
205  line.setP0( p );
206  outcode0 = r.getZone( line.getP0() );
207  } else {
208  line.setP1( p );
209  outcode1 = r.getZone( line.getP1() );
210  }
211  }
212  }
213  return accept;
214  }
