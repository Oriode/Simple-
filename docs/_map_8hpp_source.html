402 
403  quicksortStackItem * stack = new quicksortStackItem[( end - start ) / 2 + 1];
404 
405

407  stack[0].start = start;

409 
410  //initialize stack iterator
411  quicksortStackItem * top = &( stack[0] );
412 
413  // Keep popping from stack while is not empty
414  while ( top >= stack ) {
415  end = top -> end;
